# Установите Capacitor CLI и основные пакеты
npm install @capacitor/cli @capacitor/core
npm install @capacitor/android @capacitor/ios

# Инициализация Capacitor в проекте
npx cap init [название_приложения] [идентификатор_приложения] --web-dir=out

# Сборка проекта
npm run build

# Добавьте Android
npx cap add android

# Добавьте iOS
npx cap add ios

# Копирование веб-ресурсов
npx cap copy

# Обновление нативного кода
npx cap update

# Открытие проекта в нативной IDE
npx cap open android
# Для iOS (открывает Xcode)
npx cap open ios
